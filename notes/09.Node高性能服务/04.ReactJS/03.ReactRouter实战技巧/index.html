function renderRoutes(params: {
    routes: RouteProps[];
    switchProps?: SwitchProps;
}) {
    const { switchProps, routes } = params;
    return (
        <Switch {...switchProps}>
            {routes.map((route, index) => (
                <Route
                    key={index}
                    path={route.path}
                    component={route.component}
                    exact={route.exact || true}
                    strict={route.strict || false}
                ></Route>
            ))}
        </Switch>
    );
}
